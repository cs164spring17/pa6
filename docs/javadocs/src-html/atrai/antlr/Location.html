001package atrai.antlr;
002
003/**
004 * Represents a location in a source code file.
005 *
006 * @author Koushik Sen
007 * @author Alex Reinking
008 */
009public class Location {
010    public final int startLine;
011    public final int startColumn;
012    public final int endLine;
013    public final int endColumn;
014
015    /**
016     * Creates a new Location object
017     * @param startLine start line number
018     * @param startColumn start column number
019     * @param endLine end line number
020     * @param endColumn end column number
021     */
022    public Location(int startLine, int startColumn, int endLine, int endColumn) {
023        this.startLine = startLine;
024        this.startColumn = startColumn;
025        this.endLine = endLine;
026        this.endColumn = endColumn;
027    }
028
029    @Override
030    public String toString() {
031        return String.format("[begin line: %d, begin column: %d, end line: %d, end column: %d]",
032                startLine, startColumn, endLine, endColumn);
033    }
034}
