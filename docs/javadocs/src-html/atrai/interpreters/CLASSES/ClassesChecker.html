001package atrai.interpreters.CLASSES;
002
003import atrai.antlr.ANTLRTokenizer;
004import atrai.antlr.GenericAntlrToUntypedTree;
005import atrai.antlr.Location;
006import atrai.core.Pattern;
007import atrai.core.Transformer;
008import atrai.core.UntypedTree;
009import atrai.interpreters.common.Environment;
010import atrai.interpreters.common.Interpreter;
011import atrai.interpreters.common.SemanticException;
012
013import java.util.Iterator;
014
015import static atrai.interpreters.common.DynamicTypeChecker.*;
016
017public class ClassesChecker extends Interpreter {
018    private final String grammarName = "atrai.antlr.CLASSES";
019    private final ANTLRTokenizer tokenizer = new ANTLRTokenizer(grammarName);
020    private final Pattern typeExtractor = Pattern.parse("(%CLS @ typed @ @_%)", tokenizer);
021    private String currentClassName = null;
022    private String currentMethodName = null;
023    private Environment currentEnv = null;
024
025    public UntypedTree parseFile(String pgm) {
026        GenericAntlrToUntypedTree p = new GenericAntlrToUntypedTree();
027        return p.parseFileToUntypedTree("CLS", grammarName, "prog", pgm);
028    }
029
030    public Object interpret(UntypedTree st) {
031        try {
032            return null;
033        } catch (SemanticException ex) {
034            String location = ex.getLocation() != null ? " at " + ex.getLocation() : "";
035            System.err.println("Fatal Error: " + ex.getMessage() + location);
036            throw ex;
037        } catch (RuntimeException e) {
038            System.err.println(e.getMessage());
039            e.printStackTrace();
040            throw e;
041        }
042    }
043
044    public UntypedTree parseString(String pgm) {
045        GenericAntlrToUntypedTree p = new GenericAntlrToUntypedTree();
046        return p.parseStringToUntypedTree("CLS", grammarName, "prog", pgm);
047    }
048
049
050    private String extractType(Object source) {
051        Object[] tmp = typeExtractor.match(source);
052        return (String) tmp[1];
053    }
054}
