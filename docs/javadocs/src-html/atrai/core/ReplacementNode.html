006abstract public class ReplacementNode extends TreeNode {
007    int index;
008
009    /**
010     * Replace {@link ReplacementToken} holes in the tree with the capture groups
011     *
012     * @param captures the capture groups
013     * @return the capture to be used for creating the replacement
014     */
015    @Override
016    Object replace(Object[] captures) {
017        if (index < 0 || index > captures.length-1) {
018            throw new RuntimeException("Replacement failed for " + this+". No capture at index "+index);
019        }
020        return captures[index];
021
022    }
023}
