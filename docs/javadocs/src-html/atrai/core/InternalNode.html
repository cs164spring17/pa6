006/**
007 * Objects of type {@link InternalNode} forms the internal nodes of untyped trees, patterns, and templates.
008 * An InternalNode is produced by a non-terminal in a grammar. It can have multiple untyped children.
009 *
010 * @author Koushik Sen
011 * @author Alex Reinking
012 */
013public class InternalNode extends TreeNode {
014    private Object[] children;
015    private ObjectArrayList<Object> tmpChildren;
016    private boolean hasStar;
017
018    /**
019     * Create a new internal node with an unspecified number of children
020     */
021    InternalNode() {
022        tmpChildren = new ObjectArrayList<>();
023        this.hasStar = false;
024    }
025
026    /**
027     * Create a new internal node with the number of children pre-allocated.
028     *
029     * @param nChildren The number of children this node will have
030     */
031    private InternalNode(int nChildren) {
032        this.children = new Object[nChildren];
033        this.hasStar = false;
034    }
035
036    static void printTerminal(StringBuilder sb, Object child) {
037        if (child instanceof WildcardToken) {
038            sb.append(child.toString());
039        } else if (child instanceof WildcardCaptureToken) {
040            sb.append(child.toString());
041        } else if (child instanceof WildcardStarCaptureToken) {
042            sb.append(child.toString());
043        } else if (child instanceof ReplacementToken) {
044            sb.append(child.toString());
045        } else if (child instanceof ReplacementFlattenToken) {
046            sb.append(child.toString());
047        } else if (child instanceof ReplacementStarToken) {
048            sb.append(child.toString());
049        } else {
050            sb.append(SerializedTreeParser.escapeString(child.toString()));
051        }
052    }
053
054    /**
055     * Recursively visits each {@link InternalNode} among its children and replaces
056     * the {@link #children} array with the values stored in {@link #tmpChildren}
057     */
058    void finalizeAST() {
059        finalizeNode();
060        for (Object child : this.children) {
061            if (child instanceof InternalNode) {
062                ((InternalNode) child).finalizeAST();
063            }
064        }
065    }
066
067    /**
068     * Determines whether another tree rooted at an {@link InternalNode} matches this {@link Pattern} tree.
069     * It compares the structure of the two trees, using {@link TreeNode#matches(Object, ObjectArrayList)}
070     * for {@link TreeNode}s and {@link Object#equals(Object)} for other types.
071     * The other nodes populate {@code captures}
072     *
073     * @param other    The tree to compare this tree to.
074     * @param captures Populated by the matching behavior of {@code other}
075     * @return True if the trees matched, false otherwise
076     */
077    boolean matches(Object other, ObjectArrayList<Object> captures) {
078        if (!(other instanceof InternalNode)) {
079            return false;
080        }
081
082        InternalNode otherTmp = (InternalNode) other;
083        if (this.hasStar) {
084            if (this.children.length > otherTmp.children.length + 1) {
085                return false;
086            }
087        } else if (this.children.length != otherTmp.children.length) {
088            return false;
089        }
090
091        for (int i = 0; i < this.children.length; i++) {
092            if (children[i] instanceof WildcardStarCaptureToken) {
093                int len = otherTmp.children.length - this.children.length + 1;
094                ((WildcardStarCaptureToken) children[i]).match(otherTmp.children, i, len, captures);
095                i = i + len;
096            } else if (children[i] instanceof WildcardStarToken) {
097                int len = otherTmp.children.length - this.children.length + 1;
098                i = i + len;
099            } else if (children[i] instanceof TreeNode) {
100                if (!((TreeNode) children[i]).matches(otherTmp.children[i], captures)) {
101                    return false;
102                }
103            } else if (!children[i].equals(otherTmp.children[i])) {
104                return false;
105            }
106        }
107
108        return true;
109    }
110
111    /**
112     * Returns a new tree from calling {@link TreeNode#replace(Object[])} on all appropriate
113     * parts of the tree. Deep copies of non-node objects are not made.
114     *
115     * @param captures Used by {@link ReplacementToken} or custom classes to
116     *                 replace parts of the tree with elements from this array.
117     * @return The new root of the tree.
118     */
119    InternalNode replace(Object[] captures) {
120        boolean flag = false;
121        for (Object child : this.children) {
122            if (child instanceof ReplacementFlattenToken || child instanceof ReplacementStarToken) {
123                flag = true;
124            }
125        }
126        if (flag) {
127            InternalNode ret = new InternalNode();
128            for (Object o : this.children) {
129                if (o instanceof TreeNode) {
130                    Object tmp = ((TreeNode) o).replace(captures);
131                    if (o instanceof ReplacementFlattenToken) {
132                        if (!(tmp instanceof InternalNode)) {
133                            throw new RuntimeException(o + " needs to be replaced with the children of an untyped tree and cannot be replaced with " + tmp);
134                        } else {
135                            InternalNode tmp2 = (InternalNode) tmp;
136                            for (int j = 0; j < tmp2.children.length; j++) {
137                                ret.addChild(tmp2.children[j]);
138                            }
139                        }
140                    } else if (o instanceof ReplacementStarToken) {
141                        if (!(tmp instanceof Object[])) {
142                            throw new RuntimeException(o + " needs to be replaced with an array of objects and cannot be replaced with " + tmp);
143                        } else {
144                            Object[] tmp2 = (Object[]) tmp;
145                            for (Object aTmp2 : tmp2) {
146                                ret.addChild(aTmp2);
147                            }
148                        }
149                    } else {
150                        ret.addChild(tmp);
151                    }
152                } else {
153                    ret.addChild(o);
154                }
155            }
156            ret.finalizeNode();
157            return ret;
158        } else {
159            InternalNode ret = new InternalNode(this.children.length);
160            for (int i = 0; i < this.children.length; i++) {
161                Object o = this.children[i];
162                if (o instanceof TreeNode) {
163                    Object tmp = ((TreeNode) o).replace(captures);
164                    ret.children[i] = tmp;
165                } else {
166                    ret.children[i] = o;
167                }
168            }
169            return ret;
170
171        }
172    }

174    /**
175     * Prints the tree into a {@link StringBuilder}
176     *
177     * @param sb The {@link StringBuilder} to populate.
178     */
179    void toSourceString(StringBuilder sb) {
180        for (int i = 0; i < this.children.length; i++) {
181            if (i != 0) {
182                sb.append(' ');
183            }
184            Object child = this.children[i];
185            if (child instanceof InternalNode) {
186                sb.append(SerializedTreeParser.LBS);
187                ((InternalNode) child).toSourceString(sb);
188                sb.append(SerializedTreeParser.RBS);
189            } else {
190                printTerminal(sb, child);
191            }
192        }
193    }
194
195    /**
196     * Pretty prints the tree into a {@link StringBuilder}
197     *
198     * @param sb     The {@link StringBuilder} to populate.
199     * @param indent The sequence to use as indentation (tabs or some amount of spaces)
200     */
201    public void toIndentedString(StringBuilder sb, String indent) {
202        int START = 3;
203        for (int i = 0; i < this.children.length; i++) {
204            if (i != 0) {
205                sb.append(' ');
206            }
207            boolean inline1 = this.children.length == START + 1 && !(this.children[START] instanceof InternalNode);
208            if (!inline1 && i == START) {
209                sb.append("\n").append(indent);
210            }
211            Object child = this.children[i];
212            if (child instanceof InternalNode) {
213                InternalNode tmp = ((InternalNode) child);
214                boolean skip = tmp.children.length == START + 1 && !(tmp.children[START] instanceof InternalNode);
215                if (!skip && i > START) {
216                    sb.append("\n").append(indent);
217                }
218                sb.append(SerializedTreeParser.LBS);
219                tmp.toIndentedString(sb, indent + INDENTATION);
220                //if (!skip) sb.append("\n").append(indent);
221                sb.append(SerializedTreeParser.RBS);
222                if (!skip && i < this.children.length - 1) sb.append("\n").append(indent);
223            } else {
224                printTerminal(sb, child);
225            }
226
227            if (!inline1 && i == children.length - 1) {
228                sb.append("\n").append(indent.substring(0, indent.length() - INDENTATION.length()));
229            }
230        }
231    }

233    /**
234     * If the node has not been finalized, this will add {@code child} after the rest of the children.
235     *
236     * @param child The child to append
237     */
238    void addChild(Object child) {
239        if (child instanceof WildcardStarCaptureToken) {
240            if (hasStar) {
241                throw new RuntimeException("An internal node can have at most one " + WildcardStarCaptureToken.instance + " or " + WildcardStarToken.instance + " children.");
242            } else {
243                hasStar = true;
244            }
245        }
246        this.tmpChildren.push(child);
247    }
248
249    private void finalizeNode() {
250        if (tmpChildren != null) {
251            children = new Object[tmpChildren.size()];
252            int i = 0;
253            for (Object tmp : tmpChildren) {
254                children[i] = tmp;
255                i++;
256            }
257            tmpChildren = null;
258        }
259    }
260
261    /**
262     * Gets the only child of the node, otherwise throws an exception if more than one child exist (or none do)
263     *
264     * @return The unique child of the node
265     * @throws RuntimeException when the node does not have a single child
266     */
267    Object getOnlyChild() {
268        if (children.length != 1) {
269            throw new RuntimeException(this + " must be a leaf or an internal node");
270        }
271        return children[0];
272    }
273
274    /**
275     * Displays the string for debugging / re-parsing later.
276     *
277     * @return a string representation
278     */
279    public String toString() {
280        StringBuilder sb = new StringBuilder();
281        sb.append(SerializedTreeParser.LBS);
282        this.toSourceString(sb);
283        sb.append(SerializedTreeParser.RBS);
284        return sb.toString();
285    }

287    /**
288     * Accept a visitor to traverse the children of this node.  Returns the modified tree.  The original tree is not
289     * modified.
290     *
291     * @param visitor the visitor to accept
292     * @param context context to be passed to the modifiers
293     */
294    InternalNode accept(Visitor visitor, Object context) {
295        InternalNode ret = this;
296        for (int i = 0; i < this.children.length; i++) {
297            Object child = this.children[i];
298            if (child instanceof InternalNode) {
299                Object tmp = visitor.visit(child, context);
300                if (tmp != this.children[i] && ret == this) {
301                    ret = new InternalNode(this.children.length);
302                    System.arraycopy(this.children, 0, ret.children, 0, i);
303                }
304                ret.children[i] = tmp;
305            } else {
306                ret.children[i] = this.children[i];
307            }
308        }
309        return ret;
310    }

322    /**
323     * Iterates the children of this {@link InternalNode} and applies {@code lambda} to each children.  The {@code lambda}
324     * gets the child as the first argument, the reduced value of the previously visited children, and the {@code environment}
325     * as arguments.  It returns the updated reduction value, which becomes the second argument for the application
326     * of the {@code lambda} on the next children.
327     *
328     * @param lambda                function applied to each children
329     * @param initialReductionValue initial reduction value
330     * @param context               the context
331     * @return return the reduction value
332     */
333    public Object iterate(Reducer lambda, Object initialReductionValue, Object context) {
334        for (Object child : children) {
335            initialReductionValue = lambda.apply(child, initialReductionValue, context);
336        }
337        return initialReductionValue;
338    }
