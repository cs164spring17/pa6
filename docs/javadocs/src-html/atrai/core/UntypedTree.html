001package atrai.core;
002
003import atrai.antlr.Location;
004
005import java.util.ArrayList;
006
007/**
008 * A wrapper for rooted trees
009 *
010 * @author Koushik Sen
011 * @author Alex Reinking
012 */
013public class UntypedTree extends Tree {
014
015    /**
016     * Construct a new untyped tree with given root and source positions.
017     *
018     * @param treeNode  the root
019     * @param locations the corresponding source locations
020     */
021    public UntypedTree(Object treeNode, ArrayList<Location> locations) {
022        super(treeNode, locations);
023    }
024
025    /**
026     * Create a new untyped tree from some source code and a lexer.
027     *
028     * @param source The source code
029     * @param lexer  The lexer
030     * @return The tree
031     */
032    public static UntypedTree parse(String source, Lexer lexer) {
033        TreeBuilder builder = SerializedTreeParser.parse(source, lexer, false, false);
034        return new UntypedTree(builder.getTree(), builder.getLocations());
035    }
036
037    /**
038     * Get the root of the tree (typically a {@link TreeNode})
039     *
040     * @return the root
041     */
042    public Object getRoot() {
043        return rootNode;
044    }
045
046}
